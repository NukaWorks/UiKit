# Introducing MouseUI for React

## Component Template

```jsx
import React from 'react';
import PropTypes, { InferProps } from 'prop-types'

export function HelloWorld({title, color} :InferProps<typeof HelloWorld.propTypes>) {
    return (
        <h1 color={color}>{{ title }}</h1>
    );
}

HelloWorld.propTypes = {
  title: PropTypes.string.isRequired,
  color: PropTypes.string
};

HelloWorld.defaultProps = {
  title: "HelloWorld !",
  color: "blue"
}
```

# Component Stories Template
```jsx
import React from 'react';

import { HelloWorld } from './HelloWorld';

export default {
  title: 'Misc/HelloWorld',
  component: HelloWorld,
  parameters: {
    layout: 'fullscreen',
  },
};

const Template = (args) => <HelloWorld {...args} />;

export const Default = Template.bind({});
Default.args = {
  title: "Hello !"
};
```
